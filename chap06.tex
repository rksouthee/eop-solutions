\chapter{Iterators}

\begin{lemma}
	$0 \leq j \leq i \wedge \func{weak\_range}(f, i) \Rightarrow \func{weak\_range}(f, j)$
\end{lemma}

\begin{proof}
	$(0 \leq k \leq j) \Rightarrow \func{successor}^k(f) \text{ is defined.}$
\end{proof}

\begin{lemma}
	$(f + n) + m = f + (n + m)$
\end{lemma}

\begin{proof}
	\begin{align*}
		f + 0 &= f\\
		f + \func{successor}(n) &= \func{successor}(\func{successor}^n(f))
	\end{align*}
\end{proof}
